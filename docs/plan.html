// registries/types.ts
export interface PluginResult {
  id:          string;
  name:        string;
  slug:        string;
  description: string;
  author:      string;
  downloads:   number;
  versions:    string[];
  source:      'modrinth' | 'hangar';
  jarUrl?:     string;
  license?:    string;
  updated:     string;
}

export interface SearchOptions {
  limit?:      number;
  gameVersion?:string;
  categories?: string[];
}

// registries/multi.ts — 同時クエリ + マージ
export async function multiSearch(
  query: string,
  opts:  SearchOptions = {}
): Promise<PluginResult[]> {
  const [modrinthRes, hangarRes] = await Promise.allSettled([
    searchModrinth(query, opts),
    searchHangar(query, opts),
  ]);

  const results: PluginResult[] = [];
  if (modrinthRes.status === 'fulfilled') results.push(...modrinthRes.value);
  if (hangarRes.status   === 'fulfilled') results.push(...hangarRes.value);

  return dedupeBySlug(results).sort(byDownloads);
}

// commands/search.ts
import { Command } from 'commander';
import ora from 'ora';
import { multiSearch } from '../registries/multi.js';
import { formatResults } from '../formatter/human.js';

export const searchCmd = new Command('search')
  .argument('<query>', '検索キーワード')
  .option('-l, --limit <n>', '件数', '10')
  .option('-j, --json', 'JSON 出力')
  .action(async (query, opts) => {
    const spinner = ora(`Searching for "${query}"…`).start();
    try {
      const results = await multiSearch(query, { limit: +opts.limit });
      spinner.succeed(`Found ${results.length} plugins`);
      opts.json
        ? console.log(JSON.stringify(results, null, 2))
        : formatResults(results);
    } catch (e) {
      spinner.fail(String(e));
      process.exit(1);
    }
  });

// tests/multi.test.ts
import { describe, it, expect, vi } from 'vitest';
import { multiSearch } from '../src/registries/multi.js';

describe('multiSearch', () => {
  it('deduplicates results by slug', async () => {
    const results = await multiSearch('luckperms');
    const slugs = results.map(r => r.slug);
    expect(new Set(slugs).size).toBe(slugs.length);
  });

  it('sorts by downloads descending', async () => {
    const results = await multiSearch('permissions');
    for (let i = 1; i < results.length; i++) {
      expect(results[i-1].downloads).toBeGreaterThanOrEqual(results[i].downloads);
    }
  });
});
